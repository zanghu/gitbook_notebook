#include "simple_wr.h"

#define WRITE_LOG(...) \
{ \
    char buf[512]; \
    if (snprintf(buf, 512, __VA_ARGS__) >= 512) { \
        fprintf(stderr, "LOG BUFFEROVERFLOW\n"); \
    } \
    if (writeStringToPath("log.txt", buf, "a") != 0) { \
        fprintf(stderr, "WRITE_LOG failed, error.\n"); \
    } \
}
